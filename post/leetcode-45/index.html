class Solution:
    def jump(self, nums: list[int]) -> int:

        farthest, border, detect = 0, 0, 0
        for cur_pos in range(len(nums) - 1):

            if cur_pos <= farthest:
                farthest = max(farthest, cur_pos + nums[cur_pos])
                
                if cur_pos == border:
                    border = farthest
                    detect += 1

        return detect
